📋 FASE 3: MÓDULO DE PROCESSOS

3.1 Estrutura de Dados - Processo

TypeScript


interface Processo {
  id: string;
  codigo_interno: string; // Ex: "PRO.0000001"
  numero_cnj: string; // Ex: "0806008-09.2025.8.19.0024"
  
  // Partes
  cliente: {
    nome: string;
    tipo: 'pessoa_fisica' | 'pessoa_juridica';
    cpf_cnpj?: string;
  };
  envolvidos: Array<{
    nome: string;
    tipo: 'autor' | 'reu' | 'terceiro' | 'executado' | 'exequente';
  }>;
  
  // Classificação
  assunto: string;
  area_direito: string;
  
  // Localização
  orgao: string; // Ex: "TJRJ"
  comarca: string; // Ex: "Itaguaí"
  orgao_julgador?: string; // Ex: "2ª Vara Cível"
  instancia: '1ª' | '2ª' | 'TS';
  
  // Status e Controle
  status: Array<'ativo' | 'incompleto' | 'movimentado' | 'parado' | 'arquivado'>;
  valor_acao?: number;
  data_distribuicao: Date;
  
  // Organização
  responsaveis: Array<{
    usuario_id: string;
    nome: string;
    iniciais: string;
    cor: string;
  }>;
  marcadores: Array<{
    nome: string;
    cor: string;
  }>;
  
  // Caso (apelido do processo)
  caso?: {
    nome: string;
    cor: string;
  };
  
  // Metadados
  created_at: Date;
  updated_at: Date;
}

interface Tarefa {
  id: string;
  codigo_interno: string; // Ex: "TAR.0000001"
  
  // Identificação
  titulo: string;
  descricao?: string;
  tipo: 'processo' | 'revisao' | 'intimacao' | 'administrativa';
  
  // Status
  status: 'pendente' | 'a_confirmar' | 'em_execucao' | 'revisao' | 'concluida';
  
  // Prazos
  data_criacao: Date;
  data_fatal: Date; // Prazo limite
  
  // Vinculação
  processo_id?: string;
  intimacao_id?: string;
  cliente?: string;
  
  // Responsáveis
  responsaveis: Array<{
    usuario_id: string;
    nome: string;
    iniciais: string;
    cor: string;
  }>;
  criado_por: string;
  
  // Origem
  origem?: string; // Ex: "Diário de Justiça: Justiça Estadual"
  
  // Metadados
  created_at: Date;
  updated_at: Date;
}
